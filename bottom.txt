​شـرح تـركـيـب الازرار الـمـوقـع هـيـا
​أولاً: الـتـجـهـيـز (قـبـل الـعـمـل)
​قبل ما تبدأ، لازم تعرف إنك حتستخدم "الكونسول" (Console) بتاع المتصفح، وده المكان الـبـنـدك فـيـهـو الأوامـر المباشرة للموقع.
​ثـانـيـاً: كـود الـجـعـب الـمـكـرب

// كود جعب الأزرار كلها مرة واحدة
let allButtons = document.querySelectorAll('button, a, input[type="button"]');

allButtons.forEach((button, index) => {
    setTimeout(() => {
        console.log("%c جاري جعب الزر رقم: " + index, "color: green; font-weight: bold;");
        button.click();
        button.style.border = "3px solid red"; // عشان تعرف الـتـجـعـب ياتو واحد
    }, index * 200); // تأخير بسيط عشان المتصفح ما يقطع نفسه
});

ثـالـثـاً: طـريـقـة الـتـركـيـب (خـطـوة بـخـطـوة)
​افـتـح الـمـوقـع: ادخل على الصفحة الداير تشغل فيها الأزرار.
​افـتـح الـمـهـنـدس: اضغط بيمين الماوس في أي حتة فاضية واختار آخـر خـيـار (Inspect) أو اضغط من الكيبورد على زر F12.
​ادخـل الـكـونـسـول: من القائمة الفوق الظهرت ليك، فتش كلمة اسمها Console واضغط عليها.
​لـصـق الـكـود: شيل الكود النسخته فوق ده واعمل ليهو لصق (Paste) في الخط الفاضي تحت.
​الـتـنـفـيـذ: اضغط Enter وراقـب الـمـوقـع وهو ببدأ يضغط في الأزرار براااااهو
